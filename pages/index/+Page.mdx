import fs from 'fs';
import {Link} from '../../components/Link';

export const blogPosts = fs
  .readdirSync('./pages/blog/', { withFileTypes: true })
  .filter((file) => file.isDirectory());
export const blogPostImports = await Promise.all(
  blogPosts.map(async (post) => await import(`../blog/${post.name}/+Page.mdx`))
);
